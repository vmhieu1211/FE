        // hàm với tham số đầu vào và output đầu ra 
        // hàm có input và ouput 
        // nồi cơm điện thì input là điện, nước , gạo 
        // output của nồi cơm điện là cơm 

        // soA và soB là 2 param tham số đầu vào 
        // return cuối hàm là lệnh kết thúc hàm và trả về kết quả output
        function tinhTong(soA,soB){
            var c = soA + soB;
            // output 
            return c;
        }

        // hàm có param mặc định
        function tinhTong(soA = 1,soB = 1){
            var c = soA + soB;
            // output 
            return c;
        }
